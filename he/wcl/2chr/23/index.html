1וּבַשָּׁנָה הַשְּׁבִעִית הִתְחַזַּק יְהֹויָדָע וַיִּקַּח אֶת־שָׂרֵי הַמֵּאֹות לַעֲזַרְיָהוּ בֶן־יְרֹחָם וּלְיִשְׁמָעֵאל בֶּן־יְהֹוחָנָן וְלַעֲזַרְיָהוּ בֶן־עֹובֵד וְאֶת־מַעֲשֵׂיָהוּ בֶן־עֲדָיָהוּ וְאֶת־אֱלִישָׁפָט בֶּן־זִכְרִי עִמֹּו בַבְּרִית׃ 2וַיָּסֹבּוּ בִּיהוּדָה וַיִּקְבְּצוּ אֶת־הַלְוִיִּם מִכָּל־עָרֵי יְהוּדָה וְרָאשֵׁי הָאָבֹות לְיִשְׂרָאֵל וַיָּבֹאוּ אֶל־יְרוּשָׁלִָם׃ 3וַיִּכְרֹת בָּאֵי הַשַּׁבָּת לַכֹּהֲנִים וְלַלְוִיִּם לְשֹׁעֲרֵי הַסִּפִּים׃ 5וְהַשְּׁלִשִׁית בְּבֵית הַמֶּלֶךְ וְהַשְּׁלִשִׁית בְּשַׁעַר הַיְסֹוד וְכָל־הָעָם בְּחַצְרֹות בֵּית יְהוָה׃ 6וְאַל־יָבֹוא בֵית־יְהוָה כִּי אִם־הַכֹּהֲנִים וְהַמְשָׁרְתִים לַלְוִיִּם הֵמָּה יָבֹאוּ כִּי־קֹדֶשׁ הֵמָּה וְכָל־הָעָם יִשְׁמְרוּ מִשְׁמֶרֶת יְהוָה׃ 7וְהִקִּיפוּ הַלְוִיִּם אֶת־הַמֶּלֶךְ סָבִיב אִישׁ וְכֵלָיו בְּיָדֹו וְהַבָּא אֶל־הַבַּיִת יוּמָת וִהְיוּ אֶת־הַמֶּלֶךְ בְּבֹאֹו וּבְצֵאתֹו׃ 8וַיַּעֲשׂוּ הַלְוִיִּם וְכָל־יְהוּדָה כְּכֹל אֲשֶׁר־צִוָּה יְהֹויָדָע הַכֹּהֵן וַיִּקְחוּ אִישׁ אֶת־אֲנָשָׁיו בָּאֵי הַשַּׁבָּת
