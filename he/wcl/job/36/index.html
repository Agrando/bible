1וַיֹּסֶף אֱלִיהוּא וַיֹּאמַר׃ 2כַּתַּר־לִי זְעֵיר וַאֲחַוֶּךָּ כִּי עֹוד לֶאֱלֹוהַּ מִלִּים׃ 3אֶשָּׂא דֵעִי לְמֵרָחֹוק וּלְפֹעֲלִי אֶתֵּן־צֶדֶק׃ 4כִּי־אָמְנָם לֹא־שֶׁקֶר מִלָּי תְּמִים דֵּעֹות עִמָּךְ׃ 5הֶן־אֵל כַּבִּיר וְלֹא יִמְאָס כַּבִּיר כֹּחַ לֵב׃ 6לֹא־יְחַיֶּה רָשָׁע וּמִשְׁפַּט עֲנִיִּים יִתֵּן׃ 7לֹא־יִגְרַע מִצַּדִּיק עֵינָיו וְאֶת־מְלָכִים לַכִּסֵּא וַיֹּשִׁיבֵם לָנֶצַח וַיִּגְבָּהוּ׃ 8וְאִם־אֲסוּרִים בַּזִּקִּים יִלָּכְדוּן בְּחַבְלֵי־עֹנִי׃ 9וַיַּגֵּד לָהֶם פָּעֳלָם וּפִשְׁעֵיהֶם כִּי יִתְגַּבָּרוּ׃ 10וַיִּגֶל אָזְנָם לַמּוּסָר וַיֹּאמֶר כִּי־יְשֻׁבוּן מֵאָוֶן׃ 11אִם־יִשְׁמְעוּ וְיַעֲבֹדוּ יְכַלּוּ יְמֵיהֶם בַּטֹּוב וּשְׁנֵיהֶם בַּנְּעִימִים׃ 12וְאִם־לֹא יִשְׁמְעוּ בְּשֶׁלַח יַעֲבֹרוּ וְיִגְוְעוּ כִּבְלִי־דָעַת ׃ 13וְחַנְפֵי־לֵב יָשִׂימוּ אָף לֹא יְשַׁוְּעוּ כִּי אֲסָרָם׃ 14תָּמֹת בַּנֹּעַר נַפְשָׁם וְחַיָּתָם בַּקְּדֵשִׁים׃ 15יְחַלֵּץ עָנִי בְעָנְיֹו וְיִגֶל בַּלַּחַץ אָזְנָם׃ 16וְאַף הֲסִיתְךָ ׀ מִפִּי־צָר רַחַב לֹא־מוּצָק תַּחְתֶּיהָ וְנַחַת לְחָנְךָ מָלֵא דָשֶׁן׃ 17וְדִין־רָשָׁע מָלֵאתָ דִּין וּמִשְׁפָּט יִתְמֹכוּ׃ 18כִּי־חֵמָה פֶּן־יְסִיתְךָ בְסָפֶק וְרָב־כֹּפֶר אַל־יַטֶּךָּ׃ 19הֲיַעֲרֹךְ שׁוּעֲךָ לֹא בְצָר וְכֹל מַאֲמַצֵּי־כֹחַ׃ 20אַל־תִּשְׁאַף הַלָּיְלָה לַעֲלֹות עַמִּים תַּחְתָּם׃ 21הִשָּׁמֶר אַל־תֵּפֶן אֶל־אָוֶן כִּי־עַל־זֶה בָּחַרְתָּ מֵעֹנִי׃ 22הֶן־אֵל יַשְׂגִּיב בְּכֹחֹו מִי כָמֹהוּ מֹורֶה׃ 23מִי־פָקַד עָלָיו דַּרְכֹּו וּמִי־אָמַר פָּעַלְתָּ עַוְלָה׃ 24זְכֹר כִּי־תַשְׂגִּיא פָעֳלֹו אֲשֶׁר שֹׁרְרוּ אֲנָשִׁים׃ 25כָּל־אָדָם חָזוּ־בֹו אֱנֹושׁ יַבִּיט מֵרָחֹוק׃ 26הֶן־אֵל גִּיא וְלֹא נֵדָע מִסְפַּר שָׁנָיו וְלֹא־חֵקֶר׃ 27כִּי יְגָרַע נִטְפֵי־מָיִם יָזֹקּוּ מָטָר לְאֵדֹו׃ 28אֲשֶׁר־יִזְּלוּ שְׁחָקִים יִרְעֲפוּ עֲלֵי ׀ אָדָם רָב׃ 29אַף אִם־יָבִין מִפְרְשֵׂי־עָב תְּשֻׁאֹות סֻכָּתֹו׃ 30הֵן־פָּרַשׂ עָלָיו אֹורֹו וְשָׁרְשֵׁי הַיָּם כִּסָּה׃ 31כִּי־בָם יָדִין עַמִּים יִתֶּן־אֹכֶל לְמַכְבִּיר׃ 32עַל־כַּפַּיִם כִּסָּה־אֹור וַיְצַו עָלֶיהָ בְמַפְגִּיעַ׃ 33יַגִּיד עָלָיו רֵעֹו מִקְנֶה אַף עַל־עֹולֶה׃
