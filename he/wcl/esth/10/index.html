1וַיָּשֶׂם הַמֶּלֶךְ [אֲחַשֵׁרֹשׁ כ] (אֲחַשְׁוֵרֹושׁ ׀ ק) מַס עַל־הָאָרֶץ וְאִיֵּי הַיָּם׃ 2וְכָל־מַעֲשֵׂה תָקְפֹּו וּגְבוּרָתֹו וּפָרָשַׁת גְּדֻלַּת מָרְדֳּכַי אֲשֶׁר גִּדְּלֹו הַמֶּלֶךְ הֲלֹוא־הֵם כְּתוּבִים עַל־סֵפֶר דִּבְרֵי הַיָּמִים לְמַלְכֵי מָדַי וּפָרָס׃ 3כִּי ׀ מָרְדֳּכַי הַיְּהוּדִי מִשְׁנֶה לַמֶּלֶךְ אֲחַשְׁוֵרֹושׁ וְגָדֹול לַיְּהוּדִים וְרָצוּי לְרֹב אֶחָיו דֹּרֵשׁ טֹוב לְעַמֹּו וְדֹבֵר שָׁלֹום לְכָל־זַרְעֹו׃
