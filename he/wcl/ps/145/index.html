1תְּהִלָּה לְדָוִד אֲרֹומִמְךָ אֱלֹוהַי הַמֶּלֶךְ וַאֲבָרֲכָה מְךָ לְעֹולָם וָעֶד׃ 2בְּכָל־יֹום אֲבָרֲכֶךָּ וַאֲהַלְלָה מְךָ לְעֹולָם וָעֶד׃ 3גָּדֹול יְהוָה וּמְהֻלָּל מְאֹד וְלִגְדֻלָּתֹו אֵין חֵקֶר׃ 4דֹּור לְדֹור יְשַׁבַּח מַעֲשֶׂיךָ וּגְבוּרֹתֶיךָ יַגִּידוּ׃ 5הֲדַר כְּבֹוד הֹודֶךָ וְדִבְרֵי נִפְלְאֹותֶיךָ אָשִׂיחָה׃ 6וֶעֱזוּז נֹורְאֹתֶיךָ יֹאמֵרוּ [וּגְדוּלֹּתֶיךָ כ] (וּגְדוּלָּתְךָ ק) אֲסַפְּרֶנָּה׃ 7זֵכֶר רַב־טוּבְךָ יַבִּיעוּ וְצִדְקָתְךָ יְרַנֵּנוּ׃ 8חַנּוּן וְרַחוּם יְהוָה אֶרֶךְ אַפַּיִם וּגְדָל־חָסֶד׃ 9טֹוב־יְהוָה לַכֹּל וְרַחֲמָיו עַל־כָּל־מַעֲשָׂיו׃ 10יֹודוּךָ יְהוָה כָּל־מַעֲשֶׂיךָ וַחֲסִידֶיךָ יְבָרֲכוּכָה׃ 11כְּבֹוד מַלְכוּתְךָ יֹאמֵרוּ וּגְבוּרָתְךָ יְדַבֵּרוּ׃ 12לְהֹודִיעַ ׀ לִבְנֵי הָאָדָם גְּבוּרֹתָיו וּכְבֹוד הֲדַר מַלְכוּתֹו׃ 13מַלְכוּתְךָ מַלְכוּת כָּל־עֹלָמִים וּמֶמְשֶׁלְתְּךָ בְּכָל־דֹּור וָדֹור׃ 14סֹומֵךְ יְהוָה לְכָל־הַנֹּפְלִים וְזֹוקֵף לְכָל־הַכְּפוּפִים׃ 15עֵינֵי־כֹל אֵלֶיךָ יְשַׂבֵּרוּ וְאַתָּה נֹותֵן־לָהֶם אֶת־אָכְלָם בְּעִתֹּו׃ 16פֹּותֵחַ אֶת־יָדֶךָ וּמַשְׂבִּיעַ לְכָל־חַי רָצֹון׃ 17צַדִּיק יְהוָה בְּכָל־דְּרָכָיו וְחָסִיד בְּכָל־מַעֲשָׂיו׃ 18קָרֹוב יְהוָה לְכָל־קֹרְאָיו לְכֹל אֲשֶׁר יִקְרָאֻהוּ בֶאֱמֶת׃ 19רְצֹון־יְרֵאָיו יַעֲשֶׂה וְאֶת־שַׁוְעָתָם יִשְׁמַע וְיֹושִׁיעֵם׃ 20שֹׁומֵר יְהוָה אֶת־כָּל־אֹהֲבָיו וְאֵת כָּל־הָרְשָׁעִים יַשְׁמִיד׃ 21תְּהִלַּת יְהוָה יְדַבֶּר־פִּי וִיבָרֵךְ כָּל־בָּשָׂר שֵׁם קָדְשֹׁו לְעֹולָם וָעֶד׃
