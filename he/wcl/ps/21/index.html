1לַמְנַצֵּחַ מִזְמֹור לְדָוִד׃ 2יְהוָה בְּעָזְּךָ יִשְׂמַח־מֶלֶךְ וּבִישׁוּעָתְךָ מַה־יָּגֶיל מְאֹד׃ 3תַּאֲוַת לִבֹּו נָתַתָּה לֹּו וַאֲרֶשֶׁת פָתָיו בַּל־מָנַעְתָּ סֶּלָה׃ 4כִּי־תְקַדְּמֶנּוּ בִּרְכֹות טֹוב תָּשִׁית לְרֹאשֹׁו עֲטֶרֶת פָּז׃ 5חַיִּים ׀ שָׁאַל מִמְּךָ נָתַתָּה לֹּו אֹרֶךְ יָמִים עֹולָם וָעֶד׃ 6גָּדֹול כְּבֹודֹו בִּישׁוּעָתֶךָ הֹוד וְהָדָר תְּשַׁוֶּה עָלָיו׃ 7כִּי־תְשִׁיתֵהוּ בְרָכֹות לָעַד תְּחַדֵּהוּ בְשִׂמְחָה אֶת־פָּנֶיךָ׃ 8כִּי־הַמֶּלֶךְ בֹּטֵחַ בַּיהוָה וּבְחֶסֶד עֶלְיֹון בַּל־יִמֹּוט׃ 9תִּמְצָא יָדְךָ לְכָל־אֹיְבֶיךָ יְמִינְךָ תִּמְצָא שֹׂנְאֶיךָ׃ 10תְּשִׁיתֵמֹו ׀ כְּתַנּוּר אֵשׁ לְעֵת פָּנֶיךָ יְהוָה בְּאַפֹּו יְבַלְּעֵם וְתֹאכְלֵם אֵשׁ׃ 11פִּרְיָמֹו מֵאֶרֶץ תְּאַבֵּד וְזַרְעָם מִבְּנֵי אָדָם׃ 12כִּי־נָטוּ עָלֶיךָ רָעָה חָשְׁבוּ מְזִמָּה בַּל־יוּכָלוּ׃ 13כִּי תְּשִׁיתֵמֹו שֶׁכֶם בְּמֵיתָרֶיךָ תְּכֹונֵן עַל־פְּנֵיהֶם׃ 14רוּמָה יְהוָה בְעֻזֶּךָ נָשִׁירָה וּנְזַמְּרָה גְּבוּרָתֶךָ׃
