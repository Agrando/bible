1מַדּוּעַ מִשַּׁדַּי לֹא־נִצְפְּנוּ עִתִּים וְיָדְעוּ לֹא־חָזוּ יָמָיו ׃ 2גְּבֻלוֹת יַשִּׂיגוּ עֵדֶר גָּזְלוּ וַיִּרְעוּ ׃ 3חֲמוֹר יְתוֹמִים יִנְהָגוּ יַחְבְּלוּ שׁוֹר אַלְמָנָה ׃ 4יַטּוּ אֶבְיוֹנִים מִדָּרֶךְ יַחַד חֻבְּאוּ עֲנִיֵּי־אָרֶץ ׃ 5הֵן פְּרָאִים בַּמִּדְבָּר יָצְאוּ בְּפָעֳלָם מְשַׁחֲרֵי לַטָּרֶף עֲרָבָה לוֹ לֶחֶם לַנְּעָרִים ׃ 6בַּשָּׂדֶה בְּלִילוֹ יַקְצִירוּ וְכֶרֶם רָשָׁע יְלַקֵּשׁוּ ׃ 7עָרוֹם יָלִינוּ מִבְּלִי לְבוּשׁ וְאֵין כְּסוּת בַּקָּרָה ׃ 8מִזֶּרֶם הָרִים יִרְטָבוּ וּמִבְּלִי מַחְסֶה חִבְּקוּ־צוּר ׃ 9יִגְזְלוּ מִשֹּׁד יָתוֹם וְעַל־עָנִי יַחְבֹּלוּ ׃ 10עָרוֹם הִלְּכוּ בְּלִי לְבוּשׁ וּרְעֵבִים נָשְׂאוּ עֹמֶר ׃ 11בֵּין־שׁוּרֹתָם יַצְהִירוּ יְקָבִים דָּרְכוּ וַיִּצְמָאוּ ׃ 12מֵעִיר מְתִים יִנְאָקוּ וְנֶפֶשׁ־חֲלָלִים תְּשַׁוֵּעַ וֶאֱלוֹהַּ לֹא־יָשִׂים תִּפְלָה ׃ 13הֵמָּה הָיוּ בְּמֹרְדֵי־אוֹר לֹא־הִכִּירוּ דְרָכָיו וְלֹא יָשְׁבוּ בִּנְתִיבֹתָיו ׃ 14לָאוֹר יָקוּם רוֹצֵחַ יִקְטָל־עָנִי וְאֶבְיוֹן וּבַלַּיְלָה יְהִי כַגַּנָּב ׃ 15וְעֵין נֹאֵף שָׁמְרָה נֶשֶׁף לֵאמֹר לֹא־תְשׁוּרֵנִי עָיִן וְסֵתֶר פָּנִים יָשִׂים ׃ 16חָתַר בַּחֹשֶׁךְ בָּתִּים יוֹמָם חִתְּמוּ־לָמוֹ לֹא־יָדְעוּ אוֹר ׃ 17כִּי יַחְדָּו בֹּקֶר לָמוֹ צַלְמָוֶת כִּי־יַכִּיר בַּלְהוֹת צַלְמָוֶת ׃ 18קַל־הוּא עַל־פְּנֵי־מַיִם תְּקֻלַּל חֶלְקָתָם בָּאָרֶץ לֹא־יִפְנֶה דֶּרֶךְ כְּרָמִים ׃ 19צִיָּה גַם־חֹם יִגְזְלוּ מֵימֵי־שֶׁלֶג שְׁאוֹל חָטָאוּ ׃ 20יִשְׁכָּחֵהוּ רֶחֶם מְתָקוֹ רִמָּה עוֹד לֹא־יִזָּכֵר וַתִּשָּׁבֵר כָּעֵץ עַוְלָה ׃ 21רֹעֶה עֲקָרָה לֹא תֵלֵד וְאַלְמָנָה לֹא יְיֵטִיב ׃ 22וּמָשַׁךְ אַבִּירִים בְּכֹחוֹ יָקוּם וְלֹא־יַאֲמִין בַּחַיִּין ׃ 23יִתֶּן־לוֹ לָבֶטַח וְיִשָּׁעֵן וְעֵינֵיהוּ עַל־דַּרְכֵיהֶם ׃ 24רוֹמּוּ מְּעַט וְאֵינֶנּוּ וְהֻמְּכוּ כַּכֹּל יִקָּפְצוּן וּכְרֹאשׁ שִׁבֹּלֶת יִמָּלוּ ׃ 25וְאִם־לֹא אֵפוֹ מִי יַכְזִיבֵנִי וְיָשֵׂם לְאַל מִלָּתִי ׃ ס
