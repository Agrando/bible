1אִישׁ תּוֹכָחוֹת מַקְשֶׁה־עֹרֶף פֶּתַע יִשָּׁבֵר וְאֵין מַרְפֵּא ׃ 2בִּרְבוֹת צַדִּיקִים יִשְׂמַח הָעָם וּבִמְשֹׁל רָשָׁע יֵאָנַח עָם ׃ 3אִישׁ־אֹהֵב חָכְמָה יְשַׂמַּח אָבִיו וְרֹעֶה זוֹנוֹת יְאַבֶּד־הוֹן ׃ 4מֶלֶךְ בְּמִשְׁפָּט יַעֲמִיד אָרֶץ וְאִישׁ תְּרוּמוֹת יֶהֶרְסֶנָּה ׃ 5גֶּבֶר מַחֲלִיק עַל־רֵעֵהוּ רֶשֶׁת פּוֹרֵשׂ עַל־פְּעָמָיו ׃ 6בְּפֶשַׁע אִישׁ רָע מוֹקֵשׁ וְצַדִּיק יָרוּן וְשָׂמֵחַ ׃ 7יֹדֵעַ צַדִּיק דִּין דַּלִּים רָשָׁע לֹא־יָבִין דָּעַת ׃ 8אַנְשֵׁי לָצוֹן יָפִיחוּ קִרְיָה וַחֲכָמִים יָשִׁיבוּ אָף ׃ 9אִישׁ־חָכָם נִשְׁפָּט אֶת־אִישׁ אֱוִיל וְרָגַז וְשָׂחַק וְאֵין נָחַת ׃ 10אַנְשֵׁי דָמִים יִשְׂנְאוּ־תָם וִישָׁרִים יְבַקְשׁוּ נַפְשׁוֹ ׃ 11כָּל־רוּחוֹ יוֹצִיא כְסִיל וְחָכָם בְּאָחוֹר יְשַׁבְּחֶנָּה ׃ 12מֹשֵׁל מַקְשִׁיב עַל־דְּבַר־שָׁקֶר כָּל־מְשָׁרְתָיו רְשָׁעִים ׃ 13רָשׁ וְאִישׁ תְּכָכִים נִפְגָּשׁוּ מֵאִיר־עֵינֵי שְׁנֵיהֶם יְהוָה ׃ 14מֶלֶךְ שׁוֹפֵט בֶּאֱמֶת דַּלִּים כִּסְאוֹ לָעַד יִכּוֹן ׃ 15שֵׁבֶט וְתוֹכַחַת יִתֵּן חָכְמָה וְנַעַר מְשֻׁלָּח מֵבִישׁ אִמּוֹ ׃ 16בִּרְבוֹת רְשָׁעִים יִרְבֶּה־פָּשַׁע וְצַדִּיקִים בְּמַפַּלְתָּם יִרְאוּ ׃ 17יַסֵּר בִּנְךָ וִינִיחֶךָ וְיִתֵּן מַעֲדַנִּים לְנַפְשֶׁךָ ׃ פ 18בְּאֵין חָזוֹן יִפָּרַע עָם וְשֹׁמֵר תּוֹרָה אַשְׁרֵהוּ ׃ 19בִּדְבָרִים לֹא־יִוָּסֶר עָבֶד כִּי־יָבִין וְאֵין מַעֲנֶה ׃ 20חָזִיתָ אִישׁ אָץ בִּדְבָרָיו תִּקְוָה לִכְסִיל מִמֶּנּוּ ׃ 21מְפַנֵּק מִנֹּעַר עַבְדּוֹ וְאַחֲרִיתוֹ יִהְיֶה מָנוֹן ׃ 22אִישׁ־אַף יְגָרֶה מָדוֹן וּבַעַל חֵמָה רַב־פָּשַׁע ׃ 23גַּאֲוַת אָדָם תַּשְׁפִּילֶנּוּ וּשְׁפַל־רוּחַ יִתְמֹךְ כָּבוֹד ׃ 24חוֹלֵק עִם־גַּנָּב שׂוֹנֵא נַפְשׁוֹ אָלָה יִשְׁמַע וְלֹא יַגִּיד ׃ 25חֶרְדַּת אָדָם יִתֵּן מוֹקֵשׁ וּבוֹטֵחַ בַּיהוָה יְשֻׂגָּב ׃ 26רַבִּים מְבַקְשִׁים פְּנֵי־מוֹשֵׁל וּמֵיְהוָה מִשְׁפַּט־אִישׁ ׃ 27תּוֹעֲבַת צַדִּיקִים אִישׁ עָוֶל וְתוֹעֲבַת רָשָׁע יְשַׁר־דָּרֶךְ ׃ פ
