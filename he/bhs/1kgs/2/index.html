וַיָּבֵא אֶת־עֲבָדָיו מִגַּת ׃ 41וַיֻּגַּד לִשְׁלֹמֹה כִּי־הָלַךְ שִׁמְעִי מִירוּשָׁלִַם גַּת וַיָּשֹׁב ׃ 42וַיִּשְׁלַח הַמֶּלֶךְ וַיִּקְרָא לְשִׁמְעִי וַיֹּאמֶר אֵלָיו הֲלוֹא הִשְׁבַּעְתִּיךָ בַיהוָה וָאָעִד בְּךָ לֵאמֹר בְּיוֹם צֵאתְךָ וְהָלַכְתָּ אָנֶה וָאָנָה יָדֹעַ תֵּדַע כִּי מוֹת תָּמוּת וַתֹּאמֶר אֵלַי טוֹב הַדָּבָר שָׁמָעְתִּי ׃ 43וּמַדּוּעַ לֹא שָׁמַרְתָּ אֵת שְׁבֻעַת יְהוָה וְאֶת־הַמִּצְוָה אֲשֶׁר־צִוִּיתִי עָלֶיךָ ׃ 44וַיֹּאמֶר הַמֶּלֶךְ אֶל־שִׁמְעִי אַתָּה יָדַעְתָּ אֵת כָּל־הָרָעָה אֲשֶׁר יָדַע לְבָבְךָ אֲשֶׁר עָשִׂיתָ לְדָוִד אָבִי וְהֵשִׁיב יְהוָה אֶת־רָעָתְךָ
